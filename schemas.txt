CREATE TABLE if not exists users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  nickname VARCHAR(100),
  email VARCHAR(150) UNIQUE NOT NULL,
  password VARCHAR(255) NOT NULL,
  profile_pic VARCHAR(255),
  role user_role NOT NULL DEFAULT 'student',
  register_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

create table chat(
id uuid primary key default gen_random_uuid(),
user_id uuid not null,
title varchar(255) not null,
creation_date timestamp default current_timestamp,

constraint fk_chat_user
foreign key (user_id)
references users(id)
on delete cascade
)

CREATE TABLE messages (
    id uuid PRIMARY KEY default gen_random_uuid(),
    chat_id uuid NOT NULL,
    sender VARCHAR(20) NOT NULL check(sender in('user','lumina')),
    content TEXT NOT NULL,
    send_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_message_chat
        FOREIGN KEY (chat_id)
        REFERENCES chat(id)
        ON DELETE CASCADE
);

create table subjects(
id uuid primary key default gen_random_uuid(),
name varchar(100) not null,
description text
);

CREATE TABLE exercises (
    id uuid PRIMARY KEY default gen_random_uuid(),
    subjects_id uuid NOT NULL,
    question TEXT NOT NULL,
    difficulty VARCHAR(20) NOT NULL check(difficulty in ('easy','medium','hard')),
    explication TEXT,
    correct_answer CHAR(1) NOT NULL,

    CONSTRAINT fk_exercise_subject
        FOREIGN KEY (subjects_id)
        REFERENCES subjects(id)
        ON DELETE CASCADE
);

CREATE TABLE alternatives (
    id uuid PRIMARY KEY default gen_random_uuid(),
    exercises_id uuid NOT NULL,
    letter CHAR(1) NOT NULL,
    letter_text TEXT NOT NULL,

    CONSTRAINT fk_alternative_exercise
        FOREIGN KEY (exercises_id)
        REFERENCES exercises(id)
        ON DELETE CASCADE
);

CREATE TABLE progress (
    id uuid PRIMARY KEY default gen_random_uuid(),
    users_id uuid NOT NULL,
    exercises_id uuid NOT NULL,
    resposta_usuario CHAR(1),
    is_correct BOOLEAN,
    attempts INTEGER DEFAULT 1,
    answer_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_progress_user
        FOREIGN KEY (users_id)
        REFERENCES users(id)
        ON DELETE CASCADE,

    CONSTRAINT fk_progress_exercise
        FOREIGN KEY (exercises_id)
        REFERENCES exercises(id)
        ON DELETE CASCADE
);
